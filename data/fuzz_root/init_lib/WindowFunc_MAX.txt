SELECT MAX(salary) OVER(ORDER BY hire_date) AS max_salary_over_time FROM employees;

SELECT MAX(price) OVER(PARTITION BY category) AS max_price_by_category FROM products;
SELECT MAX(score) OVER(ORDER BY exam_date ROWS BETWEEN 2 PRECEDING AND CURRENT ROW) AS rolling_max_score FROM exam_scores;
