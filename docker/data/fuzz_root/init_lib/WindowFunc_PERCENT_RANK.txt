SELECT employee_id,salary,PERCENT_RANK() OVER(ORDER BY salary) AS percent_rank_salary FROM employees;
SELECT department,PERCENT_RANK() OVER(PARTITION BY department ORDER BY salary DESC) AS percent_rank_by_department FROM employees;
SELECT employee_id,bonus,PERCENT_RANK() OVER(ORDER BY bonus DESC) AS percent_rank_bonus FROM employees;
SELECT department,PERCENT_RANK() OVER(PARTITION BY region ORDER BY sales_amount) AS percent_rank_sales_region FROM sales;
SELECT product_id,price,PERCENT_RANK() OVER(PARTITION BY category ORDER BY price ASC) AS percent_rank_price_category FROM products;
SELECT department,PERCENT_RANK() OVER(PARTITION BY department ORDER BY years_of_experience) AS percent_rank_experience_department FROM employees;
SELECT student_id,score,PERCENT_RANK() OVER(PARTITION BY class ORDER BY score DESC) AS percent_rank_score_class FROM exam_scores;
SELECT customer_id,order_total,PERCENT_RANK() OVER(PARTITION BY customer_id ORDER BY order_total DESC) AS percent_rank_order_total_customer FROM orders;
SELECT department,PERCENT_RANK() OVER(PARTITION BY department ORDER BY performance_rating) AS percent_rank_performance_rating FROM employee_reviews;
SELECT project_id,project_duration,PERCENT_RANK() OVER(ORDER BY project_duration DESC) AS percent_rank_project_duration FROM projects;
SELECT product_id,price,PERCENT_RANK() OVER(ORDER BY price DESC) AS percent_rank_price_desc FROM products;
SELECT category,PERCENT_RANK() OVER(PARTITION BY category ORDER BY release_date ASC) AS percent_rank_release_date_category FROM products;
SELECT department,PERCENT_RANK() OVER(PARTITION BY department ORDER BY hire_date ASC) AS percent_rank_hire_date_department FROM employees;
SELECT class,PERCENT_RANK() OVER(PARTITION BY class ORDER BY test_score DESC) AS percent_rank_test_score_class FROM test_results;
SELECT country,PERCENT_RANK() OVER(PARTITION BY country ORDER BY gdp DESC) AS percent_rank_gdp FROM country_economics;
SELECT item_id,PERCENT_RANK() OVER(PARTITION BY category ORDER BY rating DESC) AS percent_rank_rating_category FROM product_reviews;
SELECT team,PERCENT_RANK() OVER(PARTITION BY team ORDER BY points DESC) AS percent_rank_points_team FROM league_standings;
SELECT employee_id,PERCENT_RANK() OVER(ORDER BY last_performance_rating ASC) AS percent_rank_last_performance_rating FROM employee_performance;
SELECT movie_id,PERCENT_RANK() OVER(PARTITION BY genre ORDER BY box_office DESC) AS percent_rank_box_office_genre FROM movies;
SELECT department,PERCENT_RANK() OVER(PARTITION BY department ORDER BY project_budget DESC) AS percent_rank_project_budget_department FROM projects;
