SELECT employee_id,salary,RANK() OVER(ORDER BY salary DESC) AS salary_rank FROM employees;
SELECT department,RANK() OVER(PARTITION BY department ORDER BY hire_date ASC) AS hire_date_rank FROM employees;
SELECT employee_id,bonus,RANK() OVER(ORDER BY bonus DESC) AS bonus_rank FROM employees;
SELECT department,RANK() OVER(PARTITION BY region ORDER BY sales_amount DESC) AS sales_rank_region FROM sales;
SELECT product_id,price,RANK() OVER(PARTITION BY category ORDER BY price ASC) AS price_rank_category FROM products;
SELECT department,RANK() OVER(PARTITION BY department ORDER BY years_of_experience DESC) AS experience_rank_department FROM employees;
SELECT student_id,score,RANK() OVER(PARTITION BY class ORDER BY score DESC) AS score_rank_class FROM exam_scores;
SELECT customer_id,order_total,RANK() OVER(PARTITION BY customer_id ORDER BY order_total DESC) AS order_total_rank_customer FROM orders;
SELECT department,RANK() OVER(PARTITION BY department ORDER BY performance_rating ASC) AS performance_rank_department FROM employee_reviews;
SELECT project_id,project_duration,RANK() OVER(ORDER BY project_duration DESC) AS project_duration_rank FROM projects;
SELECT employee_id,hire_date,RANK() OVER(ORDER BY hire_date ASC) AS hire_date_rank_all FROM employees;
SELECT country,RANK() OVER(PARTITION BY continent ORDER BY population DESC) AS population_rank_continent FROM countries;
SELECT category,RANK() OVER(PARTITION BY category ORDER BY rating DESC) AS rating_rank_category FROM product_reviews;
SELECT class,RANK() OVER(PARTITION BY class ORDER BY final_exam_score DESC) AS exam_score_rank_class FROM final_grades;
SELECT team,RANK() OVER(PARTITION BY league ORDER BY points DESC) AS points_rank_league FROM standings;
SELECT movie_id,RANK() OVER(PARTITION BY genre ORDER BY release_year ASC) AS release_year_rank_genre FROM movies;
SELECT department,RANK() OVER(PARTITION BY department ORDER BY project_budget DESC) AS project_budget_rank_department FROM projects;
SELECT genre,RANK() OVER(PARTITION BY genre ORDER BY user_rating DESC) AS user_rating_rank_genre FROM books;
SELECT athlete_id,RANK() OVER(PARTITION BY sport ORDER BY medals DESC) AS medals_rank_sport FROM athlete_stats;
